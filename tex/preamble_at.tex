% Half-hanging label: half in margin, half in text
\newcommand{\verselabel}[1]{%
    {\textsuperscript{\scriptsize #1}}%
}
\newcommand{\emptyverselabel}[1]{%
    \hspace*{-0.5\verselabelwidth}%
    {\textsuperscript{\scriptsize\textcolor{gray}{#1}}}%
}
\newcommand{\smartverse}[2]{%
\noindent
\if\relax\detokenize{#2}\relax
\emptyverselabel{#1}
\else
\verselabel{#1}
\fi
\,#2\par
}

\newcommand{\Verse}[6]{% fullRef, verse, suf, par?, text
  \ifnum#4=1\relax
    \if\relax\detokenize{#6}\relax
      % #6 is empty
      \par\verselabel{#1}{#5}%
    \else
      % #6 contains something
      \par\verselabel{#1(#6)}{#5}%
    \fi
  \else
    \ifnum#3=0\relax
    \verselabel{#2}{#5}%
    \else
    #5%
    \fi
  \fi
}
